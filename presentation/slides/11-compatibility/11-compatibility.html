Improvements with jQuery - Browser Compatibility

JavaScript

element.classList.toggle("class-name");

// usually needs to be
if (element.classList.contains("class-name")) {
  element.classList.remove("class-name");
} else {
  element.classList.add("class-name");
}
// not supported in IE
elements.forearch((element) => {
  element.classList.add("cold");
  element.classList.remove("hot");
});

jQuery

$("#element").toggleClass("class-name");
